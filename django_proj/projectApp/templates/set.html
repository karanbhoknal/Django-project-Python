Python Set


# Set is a unordered collection of item is called set
# We can check type of set
my={1,2,3,4,5}
print(type(my))

# Creating sets and Set of integer
my={1,2,3,4,5}
print(my)

# Set of Mixed Datatypes
my={101,"hello",(1,2,3)}
print(my)

# Duplicate value is not allowed
my={1,2,3,3,4,5}
print(my)

# Set can not have mutable items
my={1,2,3,[4,5]}
print(my)

# We can make set from a list
my=set([1,2,3,4])
print(my)

# Add Elements
my={1,2,3,4,5}
my.add(88)
print(my)

#Remove Elements
my={1,2,3,4,5}
my.remove(5)
print(my)
